## Image Upload

```html
<template>
    <div class="upload-page">
        <ChalkyImageUploader
            v-on:upload="handleUpload"
            v-on:progress="handleProgress"
            v-on:error="handleError"
        />

        <div v-if="uploading" class="progress">
            Uploading: {{ progress }}%
        </div>
    </div>
</template>

<script lang="ts">
import { Component, Vue } from 'vue-property-decorator';

@Component
export default class UploadPage extends Vue {
    uploading: boolean = false;
    progress: number = 0;

    handleProgress(percent: number): void {
        this.uploading = true;
        this.progress = percent;
    }

    handleUpload(media: any): void {
        this.uploading = false;
        console.log('Uploaded:', media.url);
    }

    handleError(error: Error): void {
        this.uploading = false;
        console.error('Upload failed:', error);
    }
}
</script>
```

## Multiple Images

```html
<template>
    <ChalkyImageUploader
        v-bind:multiple="true"
        v-bind:maxSize="10"
        v-on:upload="handleUpload"
    />
</template>

<script lang="ts">
@Component
export default class UploadPage extends Vue {
    handleUpload(media: any[]): void {
        media.forEach(m => {
            console.log('Uploaded:', m.url);
        });
    }
}
</script>
```


## Upload to Venue

```html
<template>
    <div class="venue-media">
        <ChalkyImageUploader
            v-on:upload="uploadToVenue"
        />
    </div>
</template>

<script lang="ts">
import { Component, Prop, Vue } from 'vue-property-decorator';
import { Model } from '@chalkysticks/sdk-venues';

@Component
export default class VenueMedia extends Vue {
    @Prop() venue!: Model.Venue;

    async uploadToVenue(media: any): Promise<void> {
        // Associate with venue
        await this.venue.media.add(media);

        // Refresh venue media
        await this.venue.media.fetch();
    }
}
</script>
```
